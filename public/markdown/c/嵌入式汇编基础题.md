十进制数转相应字符串

```
int intx = -57312;
char buffer[16];

_asm {
    LEA ESI, buffer
    MOV EAX, intx	; 被除数
    CMP EAX, 0      ; 判断正负
    JGE LAB1
    MOV [ESI], '-'	; 保存负号
    INC ESI
    NEG EAX			; 转正数
LAB1:
	MOV ECX, 10		; 最大循环
	MOV EBX, 10		; 除数，因为原数值为十进制数
NEXT1:
	XOR EDX, EDX	; 清空余数
	DIV EBX			; 做除法，商继续在EAX，余数在EDX
	PUSH EDX		; 将余数压入栈
	CMP EAX, 0  	; 判断商是否为0，即数值已经转换结束
	JE	NEXT1   	; 不为0，则继续循环
	SUB EBX, ECX	; 获取数字位数
	MOV ECX, EBX	; 赋值给ECX进行下次循环
NEXT2:
	POP EDX			; 取出余数
	ADD DL, '0'		; 转换成对应字符
	MOV [ESI], DL
	INC ESI
	LOOP NEXT2
END:
	MOV [ESI], '\0'	; 结束
}
```

求数组元素的和

```
int arr[] = {1, 2, 3, 4, 5};
int sum;

_asm {
    LEA EBX, arr
    MOV EAX, 0		; 存放和
    MOV ESI, 0		; 存放下标
    MOV ECX, 5		; 循环次数
NEXT:
    ADD EAX, [EBX + ESI * 4]
    INC ESI
    LOOP NEXT
END:
    MOV sum, EAX
}
```

数组寻址方式

double为8个字节(两个双字)，即64位

int，float为4个字节

short为2个字节

char为1个字节

```
int iarr[5] = {55, 87, -23, 89, 126};
double darr[5] = {9.8, 2.77, 3.1415926, 1.414, 1.73278};
int ival;
double dval;
_asm {
    LEA EBX, iarr
    MOV ECX, 3
    MOV EDX, [EBX + ECX * 4]	; 数组下标为3的元素
    MOV ival, EDX
    ;
    LEA ESI, darr
    LEA EDI dval
    MOV ECX, 2
    MOV EAX, [ESI + ECX * 8]	; 数组下标为2的元素的低双字
    MOV EDX, [ESI + ECX * 8 + 4]; 数组下标为2的元素的高双子
    MOV [EDI], EAX				; 保存低双字
    MOV [EDI + 4], EDX			; 保存低双字
}
```

指针变量地址赋值

```
char *p1, *p2;
char chx, chy;

_asm {
    LEA EAX, chx		; 取chx的地址
    MOV p1, EAX			; 将值送到指针变量p1

    MOV p2, EAX			; 将值送到指针变量p2

scanf("%c", p1);

    MOV ESI, p1			; 取回chx的值

}
```

判断字符串是否为16进制

```
char string[] = "adihwaildj12903133";
char varch = '%';
int flag;

_asm {
 	MOV AL, varch ; 存放要搜索的字符
    MOV ECX, 30	; 循环次数
    LEA EDI, string
NEXT:
    SCASB ; (AL与EDI相减，影响标志位)
    LOOPNZ ; 当ECX不等于0并且ZF不等于0则继续循环(即如果ECX=0或者找到了导致ZF=0则终止跳转)
    JNZ NOT_FOUND	; 找到，是十六进制
FOUND:
    MOV flag, 1
    JMP OVER
NOT_FOUND:
    MOV flag, 0
OVER:
}
```

地址段赋值

把位于F000H段开始的32个字节数据复制到开始抵制为B800:2000H的区域

```
    MOV AX, 0F000H	; 获取源段的段值
    MOV DS, AX		; DS存放源段段值
    MOV AX, 0B800H	; 获取目标段的段值
    MOV ES, AX		; AS存放目标段段值
    MOV ESI, 0
    MOV EDI, 2000H
    MOV ECX, 8		; 循环计数
NEXT:
	MOV EAX, [ESI]  ; 源数据段取4个字节到EAX，自动引用DS，偏移为ESI
	MOV [ES:EDI], EAX ; EAX中4个字节送到目标段，引用段寄存器ES，偏移值为EDI
	ADD ESI, 4	; 调整ESI，指向下一个双字存储单元
	ADD EDI, 4	; 调整EDI，指向下一个双字存储单元
	LOOP NEXT
```

求数值位的值是01

把AX寄存器中1的个数存放到DL中

```
    AND AX, AX
    JZ EXIT	; AX中没有1 则 取与之后全为0
    XOR DL, DL ; 清零
    MOV CX, 16 ;
AGAIN:
	ROL AX, 1 ; 左移一位，值进入CF
	JNC NEXT ; CF等于0则跳转
    INC DL
NEXT:
	LOOP AGAIN
EXIT:
```

数值转换

写一个宏将AX中的二进制转化为十进制无符号

```
TRANSFER MACRO
	MOV DX, 10 ; 余数不可能为10，所以让其作为栈底
    PUSH DX
AGAIN:
	XOR DX, DX
	MOV CX, 10
    DIV CX ; 获取当前数字，商在AX，余数在DX
    PUSH DX
    CMP AX, 0
    JE NEXT
    JMP AGAIN
NEXT:
	POP DX
	CMP DL, 10 ; 判断是否遇到栈底
    JZ EXIT
    ADD DL, 30H
    MOV AH, 2
    INT 21H 	; 显示DL中的字符
    JMP NEXT
EXIT:
```
